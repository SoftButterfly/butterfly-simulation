0011 %% Graficando la curva de la mariposa--------------------------------------
0012 %  Lipiando los espacios
0013 clear all
0014 close all
0015 clc
0016 
0017 %  Curva de la mariposa
0018 N = 50000;
0019 t = linspace(0,20*pi,N);
0020 x = sin(t).*(exp(cos(t))-2*cos(4*t)+sin(t/12).^5);
0021 y = cos(t).*(exp(cos(t))-2*cos(4*t)+sin(t/12).^5);
0022 x = x./max(abs(x)) ;
0023 y = y./max(abs(y)) ;
0024 
0025 Buterfly = figure ('Name','Butterfly',...
0026                    'NumberTitle','off',...
0027                    'Color', 'k',...
0028                    'Menubar','none');
0029 
0030 Axesfly = axes('Parent',Buterfly,...
0031                'DataAspectRatio',[2 1.7 2],...
0032                'Color',[0 0 0],...
0033                'View',[0 90],...
0034                'Xlim',[-1 1],...
0035                'Ylim',[-0.7 1],...
0036                'Zlim',[-1 1]);
0037            
0038 title('Butterfly''s curve',...
0039       'Color','w',...
0040       'FontSize',16);
0041   
0042 hold(Axesfly,'all');
0043 
0044 plot(x,y,'m');
