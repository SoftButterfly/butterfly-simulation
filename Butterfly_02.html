0011 %% Rotando las alas de la mariposa-----------------------------------------
0012 %  Lipiando los espacios
0013 clear all
0014 close all
0015 clc
0016 
0017 % Curva de la mariposa
0018 N = 50000;
0019 t = linspace(0,20*pi,N);
0020 x = sin(t).*(exp(cos(t))-2*cos(4*t)+sin(t/12).^5);
0021 y = cos(t).*(exp(cos(t))-2*cos(4*t)+sin(t/12).^5);
0022 x = x./max(abs(x)) ;
0023 y = y./max(abs(y)) ;
0024 
0025 %  Rotación 1
0026 the  = 0;
0027 z_al1 = abs(x)*sin(the);
0028 x_al1 = x*cos(the);
0029 y_al1 = y;
0030 
0031 Buterfly1 = figure ('Name','Butterfly - Rotación 0',...
0032                     'NumberTitle','off',...
0033                     'Color', 'w',...
0034                     'Menubar','none');
0035 
0036 Axesfly1 = axes('Parent',Buterfly1,...
0037                 'DataAspectRatio',[1 1 1],...
0038                 'View',[160 15],...
0039                 'Xlim',[-1 1],...
0040                 'Ylim',[-1 1],...
0041                 'Zlim',[-1 1]);
0042             
0043 hold(Axesfly1,'all');
0044 
0045 plot3(x_al1,y_al1,z_al1,'r');
0046 
0047 grid on;
0048 
0049 title({'$\bf Butterfly:\ Rotacion\ 1\ (\theta=0\ rad)$'},...
0050       'Interpreter', 'Latex',...
0051       'Color','k',...
0052       'HorizontalAlignment','center',...
0053       'FontSize',14);
0054 
0055 %  Rotación 2
0056 the  = pi/4;
0057 z_al2 = abs(x)*sin(the);
0058 x_al2 = x*cos(the);
0059 y_al2 = y;
0060 
0061 Buterfly2 = figure ('Name','Butterfly - Rotación 2',...
0062                     'NumberTitle','off',...
0063                     'Color', 'w',...
0064                     'Menubar','none');
0065 
0066 Axesfly2 = axes('Parent',Buterfly2,...
0067                 'DataAspectRatio',[1 1 1],...
0068                 'View',[160 15],...
0069                 'Xlim',[-1 1],...
0070                 'Ylim',[-1 1],...
0071                 'Zlim',[-1 1]);
0072             
0073 hold(Axesfly2,'all');
0074 
0075 plot3(x_al2,y_al2,z_al2,'g');
0076 
0077 grid on;
0078 
0079 title('$\bf Butterfly:\ Rotacion\ 2\ (\theta=\frac{\pi}{4}\ rad)$',...
0080       'Interpreter', 'Latex',...
0081       'Color','k',...
0082       'HorizontalAlignment','center',...
0083       'FontSize',14);
0084 
0085 %  Rotación 3
0086 the  = -pi/3;
0087 z_al3 = abs(x)*sin(the);
0088 x_al3 = x*cos(the);
0089 y_al3 = y;
0090 
0091 Buterfly3 = figure ('Name','Butterfly - Rotación 3',...
0092                     'NumberTitle','off',...
0093                     'Color', 'w',...
0094                     'Menubar','none');
0095 
0096 Axesfly3 = axes('Parent',Buterfly3,...
0097                 'DataAspectRatio',[1 1 1],...
0098                 'View',[160 15],...
0099                 'Xlim',[-1 1],...
0100                 'Ylim',[-1 1],...
0101                 'Zlim',[-1 1]);
0102             
0103 hold(Axesfly3,'all');
0104 
0105 plot3(x_al3,y_al3,z_al3,'b');
0106 
0107 grid on;
0108 
0109 title('$\bf Butterfly:\ Rotacion\ 3\ (\theta=-\frac{\pi}{3}\ rad)$',...
0110       'Interpreter', 'Latex',...
0111       'Color','k',...
0112       'HorizontalAlignment','center',...
0113       'FontSize',14);
0114

0116 Buterfly4 = figure ('Name','Butterfly - Vista Conjunta',...

0133
