0011 %% Butterfly: Animación 1 -> Aleteo ---------------------------------------
0012 %  Lipiando los espacios
0013 clear all
0014 close all
0015 clc
0016 
0017 % Curva de la mariposa
0018 N = 50000;
0019 t = linspace(0,20*pi,N);
0020 x = sin(t).*(exp(cos(t))-2*cos(4*t)+sin(t/12).^5);
0021 y = cos(t).*(exp(cos(t))-2*cos(4*t)+sin(t/12).^5);
0022 x = x./max(abs(x)) ;
0023 y = y./max(abs(y)) ;
0024 
0025 %  Aleteo de la maiposa
0026 n    = 5000;
0027 frec = 40*pi;
0028 time = linspace(0,5,n);
0029 the  = (pi/3)*sin(frec*time);
0030 x_al = x;
0031 y_al = y;
0032 z_al = zeros(size(x));
0033 
0034 %  Propiedades de la animacion
0035 Buterfly = figure ('Name','Butterfly - Animación 1',...
0036                    'NumberTitle','off',...
0037                    'Color', 'k',...
0038                    'Menubar','none');
0039 
0040 Axesfly = axes('Parent',Buterfly,...
0041                'DataAspectRatio',[1 1 1],...
0042                'View',[55 310],...
0043                'Xlim',[-1 1],...
0044                'Ylim',[-1 1],...
0045                'Zlim',[-1 1]);
0046 axis off;
0047 
0048 hold(Axesfly,'all');
0049 
0050 stop = uicontrol('style','toggle',...
0051                  'string','Detener',...
0052                  'background','white',...
0053                  'Position', [20 20 100 20]);
0054 
0055 Frame = plot3(x_al,y_al,z_al,'Color','m');
0056 
0057 title({'Butterfly: Animación 1 (Aleteo)'},...
0058        'Color','w',...
0059        'HorizontalAlignment','center',...
0060        'FontSize',14);
0061 
0062 %  Animación
0063 for i=1:n
0064     x_al = x.*cos(the(i));
0065     y_al = y;
0066     z_al = abs(x)*sin(the(i));
0067     
0068     set(Frame,...
0069         'XData',x_al,...
0070         'YData',y_al,...
0071         'ZData',z_al);
0072     
0073     if get(stop,'value')==0
0074         if(~mod(i,10))
0075             drawnow;
0076         end
0077     elseif get(stop,'value')==1
0078         break
0079     end
0080 end
0081 
0082 set(stop,...
0083     'style','pushbutton',...
0084     'string','Cerrar',...
0085     'callback','close(Buterfly)');
